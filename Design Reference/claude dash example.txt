(function initGlobe(){
  var canvas=document.getElementById('globe-canvas');
  if(!canvas)return;
  var ctx=canvas.getContext('2d');
  var W=290,H=290,CX=145,CY=145,R=120;
  var gRot=0,gT=0;

  var CITIES=[
    {lat:51.5,lng:-0.1},{lat:40.7,lng:-74},{lat:35.7,lng:139.7},
    {lat:1.3,lng:103.8},{lat:-33.9,lng:151.2},{lat:25.2,lng:55.3},{lat:22.3,lng:114.2}
  ];

  var ROUTES=[[0,1],[0,2],[0,3],[1,2],[2,3],[3,4],[0,5],[5,3],[2,6],[6,3]];

  var TRAVELERS=ROUTES.map(function(r,i){
    return{route:i,progress:(i/ROUTES.length),speed:0.0014+Math.random()*0.0009};
  });

  function toXYZ(lat,lng,r){
    var phi=(90-lat)*Math.PI/180;
    var theta=(lng+gRot*180/Math.PI)*Math.PI/180;
    return{x:r*Math.sin(phi)*Math.cos(theta),y:r*Math.cos(phi),z:r*Math.sin(phi)*Math.sin(theta)};
  }

  function norm(v){var m=Math.sqrt(v.x*v.x+v.y*v.y+v.z*v.z);return{x:v.x/m,y:v.y/m,z:v.z/m};}

  function slerp(a,b,f){
    var dot=Math.max(-1,Math.min(1,a.x*b.x+a.y*b.y+a.z*b.z));
    var omega=Math.acos(dot);
    if(Math.abs(omega)<0.001)return{x:a.x+(b.x-a.x)*f,y:a.y+(b.y-a.y)*f,z:a.z+(b.z-a.z)*f};
    var s=Math.sin(omega);
    return{x:(Math.sin((1-f)*omega)/s)*a.x+(Math.sin(f*omega)/s)*b.x,
           y:(Math.sin((1-f)*omega)/s)*a.y+(Math.sin(f*omega)/s)*b.y,
           z:(Math.sin((1-f)*omega)/s)*a.z+(Math.sin(f*omega)/s)*b.z};
  }

  function drawGlobe(){
    ctx.clearRect(0,0,W,H);

    /* Stars */
    for(var i=0;i<55;i++){
      var sx=(Math.sin(i*137.5)*0.5+0.5)*W;
      var sy=(Math.cos(i*97.3)*0.5+0.5)*H;
      if(Math.hypot(sx-CX,sy-CY)>R+10){
        var br=0.08+0.35*(Math.sin(gT*0.4+i)*0.5+0.5);
        ctx.beginPath();ctx.arc(sx,sy,0.7,0,Math.PI*2);
        ctx.fillStyle='rgba(200,220,255,'+br+')';ctx.fill();
      }
    }

    /* Atmosphere */
    var atm=ctx.createRadialGradient(CX,CY,R-2,CX,CY,R+20);
    atm.addColorStop(0,'rgba(78,205,180,0.14)');
    atm.addColorStop(0.5,'rgba(139,168,212,0.07)');
    atm.addColorStop(1,'rgba(139,168,212,0)');
    ctx.beginPath();ctx.arc(CX,CY,R+20,0,Math.PI*2);
    ctx.fillStyle=atm;ctx.fill();

    /* Ocean */
    var ocean=ctx.createRadialGradient(CX-22,CY-22,8,CX,CY,R);
    ocean.addColorStop(0,'#1E3A6E');
    ocean.addColorStop(0.55,'#112050');
    ocean.addColorStop(1,'#07101E');
    ctx.beginPath();ctx.arc(CX,CY,R,0,Math.PI*2);
    ctx.fillStyle=ocean;ctx.fill();

    /* Grid clip */
    ctx.save();
    ctx.beginPath();ctx.arc(CX,CY,R,0,Math.PI*2);ctx.clip();

    /* Latitude ellipses */
    for(var lat=-60;lat<=60;lat+=30){
      var phi2=(90-lat)*Math.PI/180;
      var xr2=R*Math.sin(phi2);
      ctx.beginPath();
      ctx.ellipse(CX,CY,xr2,xr2*0.15,0,0,Math.PI*2);
      ctx.strokeStyle='rgba(139,168,212,0.07)';ctx.lineWidth=0.5;ctx.stroke();
    }

    /* Longitude lines */
    for(var li=0;li<12;li++){
      var lng2=gRot+(li*Math.PI/6);
      ctx.beginPath();
      var first=true;
      for(var j=0;j<=36;j++){
        var la2=(j/36)*Math.PI-Math.PI/2;
        var lx=CX+R*Math.cos(la2)*Math.cos(lng2);
        var ly=CY-R*Math.sin(la2);
        var vis2=Math.cos(la2)*Math.cos(lng2);
        ctx.globalAlpha=vis2>0?0.09:0.02;
        if(first){ctx.moveTo(lx,ly);first=false;}else{ctx.lineTo(lx,ly);}
      }
      ctx.strokeStyle='#8BA8D4';ctx.lineWidth=0.5;ctx.stroke();
      ctx.globalAlpha=1;
    }
    ctx.restore();

    /* Arcs */
    ROUTES.forEach(function(route){
      var a=CITIES[route[0]],b=CITIES[route[1]];
      var p3a=norm(toXYZ(a.lat,a.lng,1));
      var p3b=norm(toXYZ(b.lat,b.lng,1));
      var steps=44;
      ctx.beginPath();
      var inPath=false;
      for(var s=0;s<=steps;s++){
        var arc=slerp(p3a,p3b,s/steps);
        var lx2=CX+arc.x*R*1.03;
        var ly2=CY-arc.y*R*1.03;
        var vis3=arc.z*R>-R*0.04;
        if(vis3){if(!inPath){ctx.moveTo(lx2,ly2);inPath=true;}else ctx.lineTo(lx2,ly2);}
        else inPath=false;
      }
      ctx.strokeStyle='rgba(78,205,180,0.18)';ctx.lineWidth=0.9;ctx.stroke();
    });

    /* Traveling dots */
    TRAVELERS.forEach(function(tv){
      var route=ROUTES[tv.route];
      var a=CITIES[route[0]],b=CITIES[route[1]];
      var p3a=norm(toXYZ(a.lat,a.lng,1));
      var p3b=norm(toXYZ(b.lat,b.lng,1));
      var arc=slerp(p3a,p3b,tv.progress);
      var px=CX+arc.x*R*1.03;
      var py=CY-arc.y*R*1.03;
      if(arc.z*R>-R*0.04){
        var t0=Math.max(0,tv.progress-0.07);
        var a0=slerp(p3a,p3b,t0);
        var tx2=CX+a0.x*R*1.03,ty2=CY-a0.y*R*1.03;
        var tg=ctx.createLinearGradient(tx2,ty2,px,py);
        tg.addColorStop(0,'rgba(78,205,180,0)');
        tg.addColorStop(1,'rgba(78,205,180,0.65)');
        ctx.beginPath();ctx.moveTo(tx2,ty2);ctx.lineTo(px,py);
        ctx.strokeStyle=tg;ctx.lineWidth=1.5;ctx.stroke();
        ctx.beginPath();ctx.arc(px,py,2.5,0,Math.PI*2);
        ctx.fillStyle='#4ECDB4';ctx.fill();
      }
      tv.progress+=tv.speed;
      if(tv.progress>1)tv.progress=0;
    });

    /* City markers */
    CITIES.forEach(function(city,i){
      var p3=toXYZ(city.lat,city.lng,R);
      var px=CX+p3.x,py=CY-p3.y;
      if(p3.z<-R*0.04)return;
      var pulse=Math.sin(gT*2+i*1.4)*0.5+0.5;
      ctx.beginPath();ctx.arc(px,py,4+pulse*3.5,0,Math.PI*2);
      ctx.strokeStyle='rgba(78,205,180,'+(0.12+pulse*0.14)+')';ctx.lineWidth=1;ctx.stroke();
      ctx.beginPath();ctx.arc(px,py,3.5,0,Math.PI*2);
      ctx.strokeStyle='rgba(78,205,180,0.45)';ctx.lineWidth=1;ctx.stroke();
      ctx.beginPath();ctx.arc(px,py,2,0,Math.PI*2);
      ctx.fillStyle='#4ECDB4';ctx.fill();
    });

    /* Edge ring */
    ctx.beginPath();ctx.arc(CX,CY,R,0,Math.PI*2);
    ctx.strokeStyle='rgba(139,168,212,0.18)';ctx.lineWidth=1;ctx.stroke();

    gRot+=0.006;
    gT+=0.028;
    requestAnimationFrame(drawGlobe);
  }
  drawGlobe();
})();

function md2html(t){
  if(!t)return'';
  return t.split('\n\n').map(para=>{
    const lines=para.split('\n').map(l=>l.replace(/\*\*(.+?)\*\*/g,'<b>$1</b>').replace(/\*(.+?)\*/g,'<em>$1</em>'));
    if(lines.length===1&&lines[0].match(/^\d+\./))return'<p>'+lines[0]+'</p>';
    return'<p>'+lines.join('<br>')+'</p>';
  }).join('');
}

}

async function fetchFile(name,token){
  const r=await fetch(`https://api.github.com/repos/${GH_USER}/${GH_REPO}/contents/dashboard/${name}.md`,{headers:{'Authorization':'Bearer '+token,'Accept':'application/vnd.github.v3+json'}});
  if(!r.ok)throw new Error(r.status);
  const d=await r.json();
  const bin=atob(d.content.replace(/\n/g,''));
  const bytes=new Uint8Array(bin.length);
  for(let i=0;i<bin.length;i++)bytes[i]=bin.charCodeAt(i);
  return new TextDecoder('utf-8').decode(bytes);
}
